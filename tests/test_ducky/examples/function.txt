FUNCTION PRINT_A()
    STRING a
END_FUNCTION

FUNCTION B()
    STRING b
END_FUNCTION

PRINT_A()
B()

FUNCTION C()
    RETURN TRUE
END_FUNCTION

FUNCTION D()
    PRINT_A()
    B()
    RETURN FALSE
END_FUNCTION

IF (C() == TRUE) THEN
    STRING c
END_IF

IF (D() == FALSE) THEN
    STRING d
END_IF

FUNCTION E()
    RETURN TRUE
END_FUNCTION

FUNCTION F()
    IF (E() == TRUE) THEN
        RETURN TRUE
    ELSE
        RETURN FALSE
    END_IF
END_FUNCTION

FUNCTION G()
    IF (F() == FALSE) THEN
        STRING a
    ELSE
        STRING b
    END_IF
END_FUNCTION

G()
